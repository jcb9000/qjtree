331 void JsonTreeModel::updateNode(TreeNode *node, int i, const QString oldKey, const QString newKey, const QJsonValue value)
332 {
333  QJsonObject jobj;
334  QJsonArray jarr;
335 
336

340 
341  node->key = newKey;
